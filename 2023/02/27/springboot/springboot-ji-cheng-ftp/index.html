ftp客户端工场类 FTPClientFactory

package com.gxxmt.c1.ftp;

import org.apache.commons.net.ftp.FTPClient;
import org.apache.commons.net.ftp.FTPReply;
import org.apache.commons.pool2.PooledObject;
import org.apache.commons.pool2.PooledObjectFactory;
import org.apache.commons.pool2.impl.DefaultPooledObject;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @Author longshengwen
 * @Date 2022/11/7 10:17
 * @Desc
 * @Version 1.0
 */
public class FTPClientFactory implements PooledObjectFactory<FTPClient> {

    private final Logger logger = LoggerFactory.getLogger(FTPClientFactory.class);

    private FTPClientConfig ftpConfig;

    public FTPClientFactory(FTPClientConfig ftpConfig) {
        this.ftpConfig = ftpConfig;
    }

    /**
     * 创建链接对象
     * @return
     * @throws Exception
     */
    @Override
    public PooledObject<FTPClient> makeObject() throws Exception {
        FTPClient ftpClient = new FTPClient();
        ftpClient.setControlEncoding(ftpConfig.getEncoding());
        ftpClient.setConnectTimeout(ftpConfig.getClientTimeout());
        try {
            ftpClient.connect(ftpConfig.getHost(), ftpConfig.getPort());
            int reply = ftpClient.getReplyCode();
            if (!FTPReply.isPositiveCompletion(reply)) {
                ftpClient.disconnect();
                logger.error("FTP拒绝链接！");
                return null;
            }
            boolean result = ftpClient.login(ftpConfig.getUserName(), ftpConfig.getPassword());
            ftpClient.setFileType(ftpConfig.getTransferFileType());
            if (!result) {
                logger.error("FTP登陆失败..用户名为: {}", ftpConfig.getUserName());
            }
        } catch (Exception e) {
            logger.error("创建FTP链接异常 {}", e);
            throw e;
        }

        return new DefaultPooledObject<>(ftpClient);
    }

    /**
     * 销毁链接
     * @param pooledObject
     * @throws Exception
     */
    @Override
    public void destroyObject(PooledObject<FTPClient> pooledObject) throws Exception {
        FTPClient ftpClient = pooledObject.getObject();
        try {
            if (ftpClient != null && ftpClient.isConnected()) {
                ftpClient.logout();
            }
        } catch (Exception e) {
            logger.error("FTP退出登陆异常 {}", e);
            throw e;
        } finally {
            if (ftpClient != null) {
                ftpClient.disconnect();
            }
        }
    }

    /**
     * 验证链接
     * @param pooledObject
     * @return
     */
    @Override
    public boolean validateObject(PooledObject<FTPClient> pooledObject) {
        try {
            FTPClient ftpClient = pooledObject.getObject();
            return ftpClient.sendNoOp();
        } catch (Exception e) {
            logger.error("FTP 客户端验证失败: {}", e);
        }
        return false;
    }

    /**
     * 初始化链接
     * @param pooledObject
     * @throws Exception
     */
    @Override
    public void activateObject(PooledObject<FTPClient> pooledObject) throws Exception {

    }

    /**
     * 重置链接
     * @param pooledObject
     * @throws Exception
     */
    @Override
    public void passivateObject(PooledObject<FTPClient> pooledObject) throws Exception {

    }
}

链接池自动装配 FTPClientPoolConfiguration

package com.gxxmt.c1.ftp;

import org.apache.commons.net.ftp.FTPClient;
import org.apache.commons.pool2.impl.GenericObjectPoolConfig;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import javax.annotation.PreDestroy;
import java.time.Duration;

/**
 * @Author longshengwen
 * @Date 2022/11/9 17:53
 * @Desc FTP链接池自动配置
 * @Version 1.0
 */
@Configuration
public class FTPClientPoolConfiguration {

    private FTPClientPool pool;

    @Autowired
    private FTPClientConfig ftpClientConfig;

    @Autowired
    public FTPClientPoolConfiguration() {}

    @Bean
    @ConditionalOnClass({FTPClientFactory.class})
    protected FTPClientPool ftpClientPool() {
        FTPClientFactory ftpClientFactory = new FTPClientFactory(this.ftpClientConfig);
        GenericObjectPoolConfig<FTPClient> poolConfig = new GenericObjectPoolConfig<>();
        // 链接池中最大空闲的连接数,默认为8
        poolConfig.setMaxIdle(this.ftpClientConfig.getPool().getMaxIdle());
        // 连接池中最少空闲的连接数,默认为0
        poolConfig.setMinIdle(this.ftpClientConfig.getPool().getMinIdle());
        poolConfig.setMaxTotal(this.ftpClientConfig.getPool().getMaxTotal());
        poolConfig.setBlockWhenExhausted(true);
        // 向调用者输出“链接”资源时，是否检测是有有效，如果无效则从连接池中移除，并尝试获取继续获取。
        poolConfig.setTestOnBorrow(this.ftpClientConfig.getPool().isTestOnBorrow());
        // 向连接池“归还”链接时，是否检测“链接”对象的有效性。
        poolConfig.setTestOnReturn(false);
        poolConfig.setTestWhileIdle(true);
        //  “空闲链接”检测线程，检测的周期，毫秒数
        poolConfig.setTimeBetweenEvictionRuns(Duration.ofSeconds(30));
        // 一定要关闭jmx，不然springboot启动会报已经注册了某个jmx的错误
        poolConfig.setJmxEnabled(false);

        pool = new FTPClientPool(ftpClientFactory, poolConfig);
        initPool(this.ftpClientConfig.getPool().getInitSize(), this.ftpClientConfig.getPool().getMaxIdle());
        return pool;
    }

    /**
     * 预加载链接池
     * @param initialSize
     * @param maxIdle
     */
    private void initPool(int initialSize, int maxIdle) {
        if (initialSize <= 0) {
            return;
        }
        int size = Math.min(initialSize, maxIdle);
        for (int i = 0; i < size; i++) {
            try {
                pool.addObject();
            } catch (Exception e) {
                throw new RuntimeException(e);
            }
        }
    }

    @PreDestroy
    public void destory() {
        if (pool != null) {
            pool.close();
        }
    }
}

ftp工具类 FTPUtils

package com.gxxmt.c1.ftp;

import org.apache.commons.net.ftp.FTPClient;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.io.*;
import java.util.StringTokenizer;

/**
 * @Author longshengwen
 * @Date 2022/11/7 10:56
 * @Desc
 * @Version 1.0
 */
@Component
public class FTPUtils {

    private static final Logger logger = LoggerFactory.getLogger(FTPUtils.class);

    @Autowired
    public FTPClientPool ftpClientPool;

    /**
     * 获取链接对象
     * @return
     * @throws Exception
     */
    public FTPClient getFtpClient() throws Exception {
        FTPClient ftpClient = ftpClientPool.borrowObject();
        return ftpClient;
    }

    /**
     * 返还链接至链接池
     * @param client
     */
    public void returnClient(FTPClient client) {
        if (client != null) {
            ftpClientPool.returnObject(client);
        }
    }

    /**
     * 上传文件流
     * @param remoteFile
     * @param input
     * @return
     */
    public boolean uploadFile(String remoteFile, InputStream input) {
        boolean result = false;
        FTPClient ftpClient = null;
        try {
            ftpClient = getFtpClient();
            ftpClient.enterLocalPassiveMode();
            result = ftpClient.storeFile(remoteFile, input);
            input.close();
        } catch (Exception e) {
            logger.error("FTP 上传文件失败: {}", e);
            logger.error("上传文件失败: {}", remoteFile);
        } finally {
            returnClient(ftpClient);
        }
        return result;
    }

    /**
     * 上传本地文件
     * @param remoteFile
     * @param localFile
     * @return
     */
    public boolean uploadFile(String remoteFile, String localFile) {
        FileInputStream inputStream = null;
        try {
            inputStream = new FileInputStream(new File(localFile));
        } catch (FileNotFoundException e) {
            logger.error("文件不存在：{}", localFile);
            return false;
        }
        return uploadFile(remoteFile, inputStream);
    }

    /**
     * 获取文件输入流
     * @param fileName
     * @return
     * @throws Exception
     */
    public InputStream getFileInputStream(String fileName) throws Exception {
        FTPClient ftpClient = null;
        ByteArrayInputStream inputStream = null;
        try {
            ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
            ftpClient = getFtpClient();
            ftpClient.retrieveFile(fileName, outputStream);
            inputStream = new ByteArrayInputStream(outputStream.toByteArray());
            outputStream.close();
        } catch (Exception e) {
            logger.error("FTP 获取文件{}输入流失败: {}", fileName, e);
        } finally {
            returnClient(ftpClient);
        }
        return inputStream;
    }

    /**
     * 拷贝文件
     * @param fromFile
     * @param toFile
     * @return
     * @throws Exception
     */
    public boolean copyFile(String fromFile, String toFile) throws Exception {
        InputStream inputStream = getFileInputStream(fromFile);
        FTPClient ftpClient = null;
        boolean flag = false;
        try {
            ftpClient = getFtpClient();
            flag = ftpClient.storeFile(toFile, inputStream);
            inputStream.close();
        } catch (Exception e) {
            logger.error("FTP 拷贝文件{}失败: {}", fromFile, e);
        } finally {
            returnClient(ftpClient);
        }
        return flag;
    }

    /**
     * 下载文件
     * @param remoteFile
     * @param localFile
     * @return
     */
    public  boolean downFile(String remoteFile, String localFile) {
        boolean result = false;
        FTPClient ftpClient = null;
        try {
            ftpClient = getFtpClient();
            OutputStream outputStream = new FileOutputStream(localFile);
            ftpClient.retrieveFile(remoteFile, outputStream);
            outputStream.close();
            result = true;
        } catch (Exception e) {
            logger.error("下载文件异常：{}", e);
        } finally {
            returnClient(ftpClient);
        }
        return result;
    }

    /**
     * 获取文件流
     * @param remoteFile
     * @return
     * @throws Exception
     */
    public InputStream getInputStream(String remoteFile) throws Exception {
        FTPClient ftpClient = getFtpClient();
        InputStream inputStream = ftpClient.retrieveFileStream(remoteFile);
        returnClient(ftpClient);
        return inputStream;
    }

    /**
     * 创建文件夹
     * @param remoteDir
     * @return
     * @throws Exception
     */
    public boolean makeDirectory(String remoteDir) {
        FTPClient ftpClient = null;
        boolean result = false;
        try {
            ftpClient = getFtpClient();
            result = ftpClient.makeDirectory(remoteDir);
        } catch (IOException e) {
            logger.error("FTP 创建文件夹异常 {}", e);
        } catch (Exception e) {
            logger.error("获取链接失败 {}", e);
        } finally {
            returnClient(ftpClient);
        }
        return result;
    }
    /**
     * 创建多层文件夹
     * @param dir
     * @return
     * @throws Exception
     */
    public boolean mkdirs(String dir) {
        boolean result = false;
        if (null == dir) {
            return result;
        }
        FTPClient ftpClient = null;
        try {
            ftpClient = getFtpClient();
            ftpClient.changeWorkingDirectory("/");
            StringTokenizer dirs = new StringTokenizer(dir, "/");
            String temp = null;
            while (dirs.hasMoreElements()) {
                temp = dirs.nextElement().toString();
                //创建目录
                ftpClient.makeDirectory(temp);
                //进入目录
                ftpClient.changeWorkingDirectory(temp);
                result = true;
            }
            ftpClient.changeWorkingDirectory("/");
        }  catch (IOException e) {
            logger.error("FTP 创建文件夹异常 {}", e);
            result = false;
        } catch (Exception e) {
            logger.error("获取链接失败 {}", e);
            result = false;
        } finally {
            returnClient(ftpClient);
        }
        return result;
    }
}
